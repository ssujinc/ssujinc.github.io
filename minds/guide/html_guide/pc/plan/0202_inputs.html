Input

Field Only
Field Only : Disabled
Text + Field
Text + Field : Disabled
<!-- -------------------- Html -------------------- -->
<!-- Input -->
<div class="iptBox">
    <!-- Field Only -->
    <input type="text" class="ipt_txt" placeholder="Type: Text">    
    <input type="number" class="ipt_txt" placeholder="Type: Number">
    <!-- Field Only : Disabled -->
    <input type="text" class="ipt_txt" placeholder="내용을 입력해 주세요." disabled="disabled">
</div>

<!-- Text + Field -->
<div class="iptBox">
    <label for="ipt_labelTxt">필드명</label>
    <input type="text" name="ipt_labelTxt" id="ipt_labelTxt" class="ipt_txt" placeholder="내용을 입력해 주세요.">
    <!-- [D] <label>의 for값과 <input>의 ID 값을 동일하게 작성해야 함 -->
</div>

<!-- Text + Field : Disabled -->
<div class="iptBox">
    <label for="ipt_labelTxt2">필드명</label>
    <input type="text" name="ipt_labelTxt2" id="ipt_labelTxt2" class="ipt_txt" placeholder="내용을 입력해 주세요." disabled="disabled">
    <!-- [D] <label>의 for값과 <input>의 ID 값을 동일하게 작성해야 함 -->
</div>
